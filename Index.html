  <script>

    // ─── STATE ────────────────────────────────────────────

    let STATE = {};

    let activeView = 'dashboard';

    let activeItem = null;

    let newItemData = {};


    const VIEW_FIELDS = {

      'Objectives':   ['Objective', 'Key Result', 'Progress %', 'Area', 'Notes'],

      'Tasks':        ['Task', 'Status', 'Objective', 'Area', 'Priority', 'Due Date', 'Notes'],

      'Reading List': ['Book Title', 'Author', 'Status', 'Genre', 'Rating', 'Notes'],

      'Link_Inbox':   ['Custom Title', 'URL', 'Website', 'Category', 'Status', 'Notes'],

    };


    // ─── BOOT ────────────────────────────────────────────

    window.onload = () => loadData();


    function loadData() {

      setLoading(true);

      google.script.run

        .withSuccessHandler(data => { STATE = data; setLoading(false); renderView(activeView); })

        .withFailureHandler(err => { setLoading(false); showError(err.message); })

        .getSheetData();

    }


    function reloadData() { loadData(); }

    function setLoading(on) {

      if (on) document.getElementById('main-content').innerHTML =

        '<div class="spinner-wrap"><div class="spinner"></div><p style="font-size:11px;letter-spacing:0.1em;">LOADING</p></div>';

    }

    function showError(msg) {

      document.getElementById('main-content').innerHTML =

        `<p style="text-align:center;opacity:0.4;margin-top:60px;font-size:13px;">Error: ${msg}</p>`;

    }


    // ─── NAV ─────────────────────────────────────────────

    function switchView(view) {

      activeView = view;

      document.querySelectorAll('.nav-bar button').forEach(b => b.classList.remove('active'));

      const nb = document.getElementById('nav-' + view);

      if (nb) nb.classList.add('active');

      renderView(view);

    }


    // ─── RENDER ROUTER ───────────────────────────────────

    function renderView(view) {

      document.getElementById('view-title').innerText =

        view === 'dashboard' ? 'Dashboard' : view.replace('_', ' ');

      if (view === 'dashboard') renderDashboard();

      else renderList(view);

    }


    // ─── DASHBOARD ───────────────────────────────────────

    function renderDashboard() {

      const tasks = STATE['Tasks'] || [];

      const todo = tasks.filter(t => t.Status === 'To Do' || t.Status === 'In Progress');

      const objectives = STATE['Objectives'] || [];

      const reading = STATE['Reading List'] || [];

      const currentBook = reading.find(b => b.Status === 'Reading');


      let html = '';


      // Today's tasks

      html += `<div class="section-label" style="margin-top:8px;">Today's Focus</div>`;

      if (todo.length === 0) {

        html += `<div class="card" style="padding:16px;margin-bottom:8px;opacity:0.5;font-size:13px;">All clear 🌿</div>`;

      } else {

        todo.slice(0, 5).forEach(t => {

          html += `<div class="card" style="padding:14px 16px;margin-bottom:8px;display:flex;align-items:center;gap:12px;cursor:pointer;" onclick='openSidebar(${JSON.stringify(t)}, "Tasks")'>

            <span class="material-icons" style="color:var(--accent);font-size:18px;">bolt</span>

            <div style="flex:1;">

              <div style="font-size:13px;font-weight:600;">${t.Task}</div>

              <div style="font-size:11px;opacity:0.4;">${t.Priority || ''} ${t.Area ? '· '+t.Area : ''}</div>

            </div>

            ${pillHtml(t.Status)}

          </div>`;

        });

      }


      // Objectives progress

      html += `<div class="section-label" style="margin-top:20px;">Objectives</div>`;

      const groups = {};

      objectives.forEach(o => { if(!groups[o.Objective]) groups[o.Objective]=[]; groups[o.Objective].push(o); });

      Object.entries(groups).slice(0,3).forEach(([g, items]) => {

        html += `<div class="card" style="padding:16px;margin-bottom:8px;">

          <div style="font-size:11px;font-weight:700;color:var(--accent);margin-bottom:10px;">${g}</div>`;

        items.forEach(item => {

          const p = Math.round(parseFloat(item['Progress %']) * 100) || 0;

          html += `<div style="margin-bottom:10px;cursor:pointer;" onclick='openSidebar(${JSON.stringify(item)}, "Objectives")'>

            <div style="display:flex;justify-content:space-between;font-size:12px;margin-bottom:4px;">

              <span>${item['Key Result'] || 'General'}</span><span>${p}%</span>

            </div>

            <div class="progress-bar"><div class="progress-fill" style="width:${p}%"></div></div>

          </div>`;

        });

        html += `</div>`;

      });


      // Currently reading

      if (currentBook) {

        html += `<div class="section-label" style="margin-top:20px;">Currently Reading</div>

        <div class="card" style="padding:16px;margin-bottom:8px;display:flex;gap:12px;align-items:center;cursor:pointer;" onclick='openSidebar(${JSON.stringify(currentBook)}, "Reading List")'>

          <span class="material-icons" style="color:var(--accent);">menu_book</span>

          <div>

            <div style="font-size:13px;font-weight:600;">${currentBook['Book Title']}</div>

            <div style="font-size:11px;opacity:0.5;">${currentBook.Author}</div>

          </div>

        </div>`;

      }


      document.getElementById('main-content').innerHTML = html;

    }


    // ─── LIST VIEW ───────────────────────────────────────

    function renderList(view) {

      const data = STATE[view] || [];

      const mc = document.getElementById('main-content');


      if (data.length === 0) {

        mc.innerHTML = `<p style="text-align:center;opacity:0.3;margin-top:60px;font-size:13px;">Nothing here yet.</p>`;

        return;

      }


      let html = '';


      if (view === 'Objectives') {

        const groups = {};

        data.forEach(o => { if(!groups[o.Objective]) groups[o.Objective]=[]; groups[o.Objective].push(o); });

        Object.entries(groups).forEach(([g, items]) => {

          html += `<div class="section-label" style="margin-top:16px;">${g}</div>

          <div class="card" style="padding:16px;margin-bottom:12px;">`;

          items.forEach(item => {

            const p = Math.round(parseFloat(item['Progress %']) * 100) || 0;

            html += `<div style="margin-bottom:14px;cursor:pointer;" onclick='openSidebar(${JSON.stringify(item)}, "Objectives")'>

              <div style="display:flex;justify-content:space-between;font-size:13px;margin-bottom:6px;">

                <span style="font-weight:500;">${item['Key Result'] || 'General'}</span><span style="opacity:0.5;">${p}%</span>

              </div>

              <div class="progress-bar"><div class="progress-fill" style="width:${p}%"></div></div>

            </div>`;

          });

          html += `</div>`;

        });


      } else if (view === 'Tasks') {

        ['High','Medium','Low',''].forEach(priority => {

          const group = data.filter(t => (t.Priority || '') === priority);

          if (group.length === 0) return;

          if (priority) html += `<div class="section-label" style="margin-top:16px;">${priority} Priority</div>`;

          group.forEach(item => {

            html += `<div class="card" style="padding:14px 16px;margin-bottom:8px;display:flex;align-items:center;gap:12px;cursor:pointer;" onclick='openSidebar(${JSON.stringify(item)}, "Tasks")'>

              <div style="flex:1;">

                <div style="font-size:13px;font-weight:600;${item.Status==='Done'?'text-decoration:line-through;opacity:0.4;':''}">${item.Task}</div>

                <div style="font-size:11px;opacity:0.4;margin-top:2px;">${item.Objective||''} ${item.Area?'· '+item.Area:''}</div>

              </div>

              ${pillHtml(item.Status)}

            </div>`;

          });

        });


      } else if (view === 'Reading List') {

        ['Reading','To Read','Finished'].forEach(status => {

          const group = data.filter(b => b.Status === status);

          if (group.length === 0) return;

          html += `<div class="section-label" style="margin-top:16px;">${status}</div>`;

          group.forEach(item => {

            html += `<div class="card" style="padding:14px 16px;margin-bottom:8px;display:flex;align-items:center;gap:12px;cursor:pointer;" onclick='openSidebar(${JSON.stringify(item)}, "Reading List")'>

              <span class="material-icons" style="color:var(--accent);font-size:20px;">menu_book</span>

              <div style="flex:1;">

                <div style="font-size:13px;font-weight:600;">${item['Book Title']}</div>

                <div style="font-size:11px;opacity:0.4;">${item.Author||''} ${item.Genre?'· '+item.Genre:''}</div>

              </div>

              ${item.Rating ? `<span style="font-size:12px;opacity:0.5;">${'★'.repeat(parseInt(item.Rating))}</span>` : ''}

            </div>`;

          });

        });


      } else if (view === 'Link_Inbox') {

        ['Unread','Read','Archived'].forEach(status => {

          const group = data.filter(l => (l.Status||'Unread') === status);

          if (group.length === 0) return;

          html += `<div class="section-label" style="margin-top:16px;">${status}</div>`;

          group.forEach(item => {

            const url = item.URL && item.URL !== 'https://...' ? item.URL : null;

            html += `<div class="card" style="padding:14px 16px;margin-bottom:8px;display:flex;align-items:center;gap:12px;cursor:pointer;" onclick='openSidebar(${JSON.stringify(item)}, "Link_Inbox")'>

              <span class="material-icons" style="color:var(--accent);font-size:20px;">link</span>

              <div style="flex:1;">

                <div style="font-size:13px;font-weight:600;">${item['Custom Title']||'Untitled'}</div>

                <div style="font-size:11px;opacity:0.4;">${item.Website||''} ${item.Category?'· '+item.Category:''}</div>

              </div>

              ${url ? `<a href="${url}" target="_blank" onclick="event.stopPropagation()" style="color:var(--accent);"><span class="material-icons" style="font-size:18px;">open_in_new</span></a>` : ''}

            </div>`;

          });

        });

      }


      mc.innerHTML = html;

    }


    // ─── PILLS ───────────────────────────────────────────

    function pillHtml(val) {

      if (!val) return '';

      const map = {

        'To Do': 'pill-todo', 'In Progress': 'pill-inprogress', 'Done': 'pill-done',

        'High': 'pill-high', 'Medium': 'pill-medium', 'Low': 'pill-low'

      };

      return `<span class="pill ${map[val]||''}">${val}</span>`;

    }


    // ─── SIDEBAR ─────────────────────────────────────────

    function openSidebar(item, view) {

      activeItem = { ...item, _view: view };

      document.getElementById('sidebar-view-label').innerText = view.replace('_',' ').toUpperCase();

      const fields = document.getElementById('sidebar-fields');

      fields.innerHTML = '';

      for (let key in item) {

        if (key.startsWith('_')) continue;

        fields.innerHTML += `

          <div style="margin-bottom:16px;">

            <label style="font-size:10px;font-weight:700;letter-spacing:0.1em;opacity:0.4;display:block;margin-bottom:6px;">${key.toUpperCase()}</label>

            <input class="input-field" type="text" value="${item[key] ?? ''}" onchange="activeItem['${key}']=this.value">

          </div>`;

      }

      document.getElementById('sidebar').classList.add('open');

      document.getElementById('overlay').classList.add('show');

    }


    function closeSidebar() {

      document.getElementById('sidebar').classList.remove('open');

      document.getElementById('overlay').classList.remove('show');

    }


    function saveItem() {

      const v = activeItem._view;

      const idx = STATE[v]?.findIndex(i => i._rowIndex === activeItem._rowIndex);

      if (idx > -1) STATE[v][idx] = { ...activeItem };

      google.script.run

        .withSuccessHandler(() => { closeSidebar(); renderView(v); })

        .withFailureHandler(err => alert('Save failed: ' + err.message))

        .syncAction('UPDATE', v, activeItem._rowIndex, activeItem);

    }


    function archiveItem() {

      if (!confirm('Archive this item?')) return;

      const v = activeItem._view;

      STATE[v] = STATE[v].filter(i => i._rowIndex !== activeItem._rowIndex);

      google.script.run

        .withSuccessHandler(() => { closeSidebar(); renderView(v); })

        .withFailureHandler(err => alert('Archive failed: ' + err.message))

        .syncAction('DELETE', v, activeItem._rowIndex, {});

    }


    // ─── ADD MODAL ───────────────────────────────────────

    function openModal() {

      newItemData = {};

      const view = activeView === 'dashboard' ? 'Tasks' : activeView;

      const fields = VIEW_FIELDS[view] || [];

      document.getElementById('modal-title').innerText = 'Add to ' + view.replace('_',' ');

      document.getElementById('modal-fields').innerHTML = fields.map(f => `

        <div>

          <label style="font-size:10px;font-weight:700;letter-spacing:0.1em;opacity:0.5;display:block;margin-bottom:4px;">${f.toUpperCase()}</label>

          <input class="input-field" type="text" placeholder="${f}" oninput="newItemData['${f}']=this.value">

        </div>`).join('');

      document.getElementById('add-modal').classList.add('show');

    }


    function closeModal() {

      document.getElementById('add-modal').classList.remove('show');

    }


    function submitAdd() {

      const view = activeView === 'dashboard' ? 'Tasks' : activeView;

      google.script.run

        .withSuccessHandler(() => { closeModal(); loadData(); })

        .withFailureHandler(err => alert('Add failed: ' + err.message))

        .syncAction('ADD', view, -1, newItemData);

    }

  </script>
