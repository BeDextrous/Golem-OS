        <h3 class="text-lg italic font-serif" id="modal-title">Add Item</h3>

  <script>

    let STATE = {};

    let activeView = 'Objectives';

    let activeItem = null;

    let newItemData = {};


    // ─── BOOT ──────────────────────────────────────────────

    window.onload = () => loadData();


    function loadData() {

      document.getElementById('v-content').innerHTML =

        '<div class="loading"><div class="spinner"></div><p class="text-xs font-sans tracking-widest uppercase">Loading...</p></div>';

      google.script.run

        .withSuccessHandler(data => { STATE = data; render(activeView); })

        .withFailureHandler(err => {

          document.getElementById('v-content').innerHTML =

            `<p class="text-center opacity-40 text-sm mt-20">Failed to load: ${err.message}</p>`;

        })

        .getSheetData();

    }


    function reloadData() { loadData(); }


    // ─── RENDER ────────────────────────────────────────────

    function render(view) {

      activeView = view;

      const content = document.getElementById('v-content');

      content.innerHTML = '';

      document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));

      const navBtn = document.getElementById('n-' + view);

      if (navBtn) navBtn.classList.add('active');

      document.getElementById('v-title').innerText = view.replace('_', ' ');


      let data = STATE[view] || [];

      const search = document.getElementById('search-input').value.toLowerCase();

      const hideDone = document.getElementById('hide-done').checked;


      data = data.filter(i => {

        const matchesSearch = Object.values(i).join(' ').toLowerCase().includes(search);

        const matchesStatus = hideDone && (view === 'Tasks' || view === 'Reading List')

          ? (i.Status !== 'Done' && i.Status !== 'Finished') : true;

        return matchesSearch && matchesStatus;

      });


      if (data.length === 0) {

        content.innerHTML = '<p class="text-center opacity-30 text-sm mt-20 font-sans">Nothing here yet.</p>';

        return;

      }


      if (view === 'Objectives') {

        const groups = {};

        data.forEach(o => {

          if (!groups[o.Objective]) groups[o.Objective] = [];

          groups[o.Objective].push(o);

        });

        for (let g in groups) {

          const card = document.createElement('div');

          card.className = 'card p-6 mb-4';

          card.innerHTML = `<h4 class="text-[10px] uppercase font-bold text-emerald-600 mb-4 tracking-widest">${g}</h4>`;

          groups[g].forEach(item => {

            let p = Math.round(parseFloat(item['Progress %']) * 100) || 0;

            const div = document.createElement('div');

            div.className = 'mb-4';

            div.onclick = () => openSidebar(item, view);

            div.innerHTML = `

              <div class="flex justify-between text-sm italic mb-1">

                <span>${item['Key Result'] || 'General'}</span>

                <span>${p}%</span>

              </div>

              <div class="w-full bg-emerald-50 h-1.5 rounded-full overflow-hidden">

                <div class="progress-fill" style="width:${p}%"></div>

              </div>`;

            card.appendChild(div);

          });

          content.appendChild(card);

        }

      } else {

        data.forEach(item => {

          const card = document.createElement('div');

          card.className = 'card p-5 flex justify-between items-center';

          card.onclick = () => openSidebar(item, view);

          const title = item.Task || item['Book Title'] || item['Custom Title'] || 'Item';

          const sub = item.Status || item.Area || item.Website || '';

          card.innerHTML = `

            <div>

              <p class="font-bold text-sm mb-1">${title}</p>

              <p class="text-[9px] uppercase opacity-40 font-bold">${sub}</p>

            </div>

            <span class="material-icons opacity-20 text-sm">chevron_right</span>`;

          content.appendChild(card);

        });

      }

    }


    // ─── SIDEBAR ───────────────────────────────────────────

    function openSidebar(item, view) {

      activeItem = { ...item, _view: view };

      const fields = document.getElementById('s-fields');

      fields.innerHTML = `<h2 class="text-2xl italic mb-4 font-serif">${view.replace('_',' ')}</h2>`;

      for (let key in item) {

        if (key.startsWith('_')) continue;

        fields.innerHTML += `

          <div class="space-y-1">

            <label class="text-[10px] uppercase font-bold opacity-40 font-sans">${key}</label>

            <input type="text" value="${item[key] ?? ''}"

              onchange="activeItem['${key}'] = this.value"

              class="input-ghibli">

          </div>`;

      }

      document.getElementById('sidebar').classList.add('open');

    }


    function closeSidebar() {

      document.getElementById('sidebar').classList.remove('open');

    }


    function saveChanges() {

      const v = activeItem._view;

      const idx = STATE[v].findIndex(i => i._rowIndex === activeItem._rowIndex);

      if (idx > -1) STATE[v][idx] = { ...activeItem };

      google.script.run

        .withSuccessHandler(() => { closeSidebar(); render(v); })

        .withFailureHandler(err => alert('Save failed: ' + err.message))

        .syncAction('UPDATE', v, activeItem._rowIndex, activeItem);

    }


    function deleteItem() {

      if (!confirm('Archive this item?')) return;

      const v = activeItem._view;

      STATE[v] = STATE[v].filter(i => i._rowIndex !== activeItem._rowIndex);

      google.script.run

        .withSuccessHandler(() => { closeSidebar(); render(v); })

        .withFailureHandler(err => alert('Delete failed: ' + err.message))

        .syncAction('DELETE', v, activeItem._rowIndex, {});

    }


    // ─── ADD MODAL ─────────────────────────────────────────

    const VIEW_FIELDS = {

      'Objectives':   ['Objective', 'Key Result', 'Progress %', 'Area', 'Notes'],

      'Tasks':        ['Task', 'Status', 'Objective', 'Area', 'Priority', 'Due Date', 'Notes'],

      'Reading List': ['Book Title', 'Author', 'Status', 'Genre', 'Rating', 'Notes'],

      'Link_Inbox':   ['Custom Title', 'URL', 'Website', 'Category', 'Status', 'Notes'],

    };


    function openAddModal() {

      newItemData = {};

      const fields = VIEW_FIELDS[activeView] || [];

      document.getElementById('modal-title').innerText = 'Add to ' + activeView.replace('_',' ');

      document.getElementById('modal-fields').innerHTML = fields.map(f => `

        <div class="space-y-1">

          <label class="text-[10px] uppercase font-bold opacity-40 font-sans">${f}</label>

          <input type="text" placeholder="${f}"

            oninput="newItemData['${f}'] = this.value"

            class="input-ghibli">

        </div>`).join('');

      document.getElementById('add-modal').classList.remove('hidden');

      document.getElementById('add-modal').classList.add('flex');

    }


    function closeAddModal() {

      document.getElementById('add-modal').classList.add('hidden');

      document.getElementById('add-modal').classList.remove('flex');

    }


    function commitAdd() {

      google.script.run

        .withSuccessHandler(() => { closeAddModal(); loadData(); })

        .withFailureHandler(err => alert('Add failed: ' + err.message))

        .syncAction('ADD', activeView, -1, newItemData);

    }


    function updateFilters() { render(activeView); }

  </script>
